export const ProgressTracker = () => {
  const [completed, setCompleted] = useState({
    installation: false,
    quickStart: false,
    architecture: false,
    components: false,
    development: false
  });
  
  const sections = [
    { key: 'installation', label: 'Installation Guide', link: '../getting-started/installation' },
    { key: 'quickStart', label: 'Quick Start', link: '../getting-started/quick-start' },
    { key: 'architecture', label: 'Architecture Overview', link: '../architecture/overview' },
    { key: 'components', label: 'Component Documentation', link: '../components/side-panel' },
    { key: 'development', label: 'Development Guide', link: '../development/contributing' }
  ];
  
  const toggleCompleted = (key) => {
    setCompleted(prev => ({ ...prev, [key]: !prev[key] }));
  };
  
  const completedCount = Object.values(completed).filter(Boolean).length;
  const progressPercent = (completedCount / sections.length) * 100;
  
  return (
    <div>
      <h4 style={{ margin: '0 0 15px 0', color: '#333' }}>Documentation Progress</h4>
      
      <div style={{
        background: '#e9ecef',
        borderRadius: '10px',
        height: '20px',
        marginBottom: '20px',
        overflow: 'hidden'
      }}>
        <div style={{
          background: 'linear-gradient(90deg, #28a745, #20c997)',
          height: '100%',
          width: `${progressPercent}%`,
          transition: 'width 0.3s ease',
          borderRadius: '10px'
        }} />
      </div>
      
      <div style={{ marginBottom: '10px', fontWeight: 'bold', color: '#495057' }}>
        {completedCount} of {sections.length} sections completed ({Math.round(progressPercent)}%)
      </div>
      
      {sections.map(section => (
        <div key={section.key} style={{
          display: 'flex',
          alignItems: 'center',
          gap: '10px',
          padding: '8px 0',
          borderBottom: '1px solid #e9ecef'
        }}>
          <input
            type="checkbox"
            checked={completed[section.key]}
            onChange={() => toggleCompleted(section.key)}
            style={{ transform: 'scale(1.2)' }}
          />
          <span style={{
            textDecoration: completed[section.key] ? 'line-through' : 'none',
            color: completed[section.key] ? '#6c757d' : '#333',
            flex: 1
          }}>
            {section.label}
          </span>
          <a 
            href={section.link}
            style={{
              color: '#007acc',
              textDecoration: 'none',
              fontSize: '14px'
            }}
          >
            View ‚Üí
          </a>
        </div>
      ))}
    </div>
  );
};

### Build Time Complexity

The file tree building algorithm has a time complexity of **O(n log n)** where:
- **n** = number of files in the documentation
- **log n** factor comes from the sorting operation

export const LiveFileTree = () => {
  // This would normally import the actual buildFileTree function
  // but for demo purposes, we'll simulate it
  const mockFileTree = [
    {
      name: 'README.md',
      path: '/src/docs/README.md',
      type: 'file'
    },
    {
      name: 'getting-started',
      path: '/src/docs/getting-started',
      type: 'folder',
      children: [
        { name: 'installation.md', path: '/src/docs/getting-started/installation.md', type: 'file' },
        { name: 'quick-start.md', path: '/src/docs/getting-started/quick-start.md', type: 'file' },
        { name: 'configuration.mdx', path: '/src/docs/getting-started/configuration.mdx', type: 'file' }
      ]
    },
    {
      name: 'architecture',
      path: '/src/docs/architecture',
      type: 'folder',
      children: [
        { name: 'overview.md', path: '/src/docs/architecture/overview.md', type: 'file' },
        { name: 'file-tree-system.md', path: '/src/docs/architecture/file-tree-system.md', type: 'file' },
        { name: 'routing.md', path: '/src/docs/architecture/routing.md', type: 'file' }
      ]
    }
  ];
  
  const [expandedFolders, setExpandedFolders] = useState(new Set(['getting-started']));
  
  const toggleFolder = (path) => {
    setExpandedFolders(prev => {
      const next = new Set(prev);
      if (next.has(path)) {
        next.delete(path);
      } else {
        next.add(path);
      }
      return next;
    });
  };
  
  const renderNode = (node, level = 0) => {
    const isExpanded = expandedFolders.has(node.path);
    
    return (
      <div key={node.path}>
        <div 
          style={{
            paddingLeft: `${level * 20 + 10}px`,
            padding: '5px',
            display: 'flex',
            alignItems: 'center',
            gap: '8px',
            cursor: node.type === 'folder' ? 'pointer' : 'default',
            background: level % 2 === 0 ? '#f8f9fa' : 'white'
          }}
          onClick={() => node.type === 'folder' && toggleFolder(node.path)}
        >
          <span style={{ fontSize: '14px' }}>
            {node.type === 'folder' ? (isExpanded ? 'üìÇ' : 'üìÅ') : 'üìÑ'}
          </span>
          <span style={{ fontSize: '14px', fontFamily: 'monospace' }}>
            {node.name}
          </span>
        </div>
        
        {node.type === 'folder' && isExpanded && node.children && (
          <div>
            {node.children.map(child => renderNode(child, level + 1))}
          </div>
        )}
      </div>
    );
  };
  
  return (
    <div>
      <div style={{
        background: '#343a40',
        color: 'white',
        padding: '10px 15px',
        fontWeight: 'bold'
      }}>
        Live File Tree Structure
      </div>
      <div style={{ maxHeight: '300px', overflow: 'auto' }}>
        {mockFileTree.map(node => renderNode(node))}
      </div>
    </div>
  );
};
